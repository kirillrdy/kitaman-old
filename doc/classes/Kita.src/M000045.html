  def call(action)
  
    for dependency in @depend
      Kita.new(dependency).call(action)
    end
  
    if action == :install
      download unless downloaded?

      #TODO Clean
      puts "Installing #{self.to_s}".bold.green unless installed?
      install unless installed?
    end
    
    if action == :remove
      remove if installed?
    end
    
  end
